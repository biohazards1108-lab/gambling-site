function showAdminSections() { document.getElementById("statsSection").style.display = "block"; document.getElementById("usersSection").style.display = "block"; document.getElementById("gamesSection").style.display = "block"; document.getElementById("logsSection").style.display = "block"; document.getElementById("maintenanceSection").style.display = "block"; } // --------------------------------------------- // LOAD STATS // --------------------------------------------- function loadStats() { fetch(API + "/admin/stats", { headers: { "x-admin-key": ADMIN_KEY } }) .then(res => res.json()) .then(data => { document.getElementById("statsOutput").innerText = `Total Users: ${data.totalUsers} Total Balance: $${data.totalBalance} Active Games: ${data.activeGames}`; }); } // --------------------------------------------- // LOAD USERS // --------------------------------------------- function loadUsers() { fetch(API + "/admin/users", { headers: { "x-admin-key": ADMIN_KEY } }) .then(res => res.json()) .then(users => { let html = ` ID Username Balance Banned Actions `; users.forEach(u => { html += ` ${u.id} ${u.username} $${u.balance} ${u.banned} `; }); document.getElementById("usersTable").innerHTML = html; }); } function banUser(id) { fetch(API + "/admin/ban/" + id, { method: "POST", headers: { "x-admin-key": ADMIN_KEY } }).then(() => loadUsers()); } function unbanUser(id) { fetch(API + "/admin/unban/" + id, { method: "POST", headers: { "x-admin-key": ADMIN_KEY } }).then(() => loadUsers()); } // --------------------------------------------- // TOGGLE GAME // --------------------------------------------- function toggleGame() { const game = document.getElementById("gameName").value.trim(); fetch(API + "/admin/toggle-game/" + game, { method: "POST", headers: { "x-admin-key": ADMIN_KEY } }) .then(() => { document.getElementById("gameStatus").innerText = game + " toggled!"; }); } // --------------------------------------------- // LOAD LOGS // --------------------------------------------- function loadLogs() { fetch(API + "/admin/logs", { headers: { "x-admin-key": ADMIN_KEY } }) .then(res => res.text()) .then(text => { document.getElementById("logBox").innerText = text; }); } // --------------------------------------------- // MAINTENANCE MODE // --------------------------------------------- function toggleMaintenance() { fetch(API + "/admin/maintenance", { method: "POST", headers: { "x-admin-key": ADMIN_KEY } }) .then(() => { document.getElementById("maintenanceStatus").innerText = "Maintenance mode toggled!"; }); }
